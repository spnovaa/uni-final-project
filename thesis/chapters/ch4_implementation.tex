\section{مقدمه}
%----------------------------------------------------------------------

در این فصل جزئیات پیاده‌سازی سامانه‌ی واسط هوش مصنوعی تشریح می‌شود.
هر ماژول به ترتیب وابستگی معرفی شده و برای هر بخش نمونه‌های عملی از
فراخوانی \lr{API} ارائه می‌گردد. تمامی نمونه‌ها با ابزار \lr{Postman}
بر روی نمونه‌ی محلی سامانه اجرا شده‌اند.

%----------------------------------------------------------------------
\section{فناوری‌های استفاده‌شده}
%----------------------------------------------------------------------

\subsection{چارچوب \lr{Laravel}}

\lr{Laravel} یک چارچوب \lr{PHP} با معماری \lr{MVC} است که زیرساخت قدرتمندی
برای ساخت \lr{API}های \lr{RESTful} فراهم می‌کند. استفاده از الگوی
\lr{Pipeline} داخلی \lr{Laravel} امکان پیاده‌سازی خط لوله‌ی پردازش درخواست
را با اتصال‌پذیری بالا فراهم کرده است.

\subsection{پایگاه داده‌ی \lr{SQL Server}}

\lr{Microsoft SQL Server} به عنوان پایگاه داده‌ی اصلی انتخاب شده است.
تمامی اطلاعات کاربران، کلیدهای \lr{API}، ارائه‌دهندگان، گزارش‌های مصرف
و تراکنش‌های مالی در این پایگاه داده ذخیره می‌شوند.

\subsection{\lr{Redis}}

\lr{Redis} برای دو منظور به‌کار می‌رود: کش کردن نتایج پرتکرار (مانند
پیکربندی ارائه‌دهندگان و مدل‌ها) و مدیریت صف‌های پردازش پس‌زمینه برای
ارسال رمز یک‌بارمصرف.

\subsection{\lr{Vue.js}}

\lr{Vue.js} برای پیاده‌سازی داشبورد \lr{Playground} استفاده شده است که
امکان آزمایش تعاملی نقاط پایانی سامانه را فراهم می‌کند.

%----------------------------------------------------------------------
\section{ساختار پوشه‌بندی پروژه}
%----------------------------------------------------------------------

پروژه از الگوی معماری \term{حوزه‌محور}{Domain-Driven Design} پیروی می‌کند.
کلاس‌های مربوط به هر حوزه‌ی کاری در پوشه‌ی \code{app/Domains} قرار دارند
که شامل زیرحوزه‌های احراز هویت (\code{Auth})، سامانه‌ی واسط (\code{Gateway})،
کلیدهای \lr{API} (\code{Keys}) و ارائه‌دهندگان (\code{Providers}) است.
کنترلرها در \code{app/Http/Controllers/Api/V1} و خط لوله‌های پردازش درخواست
در \code{app/Pipelines} نگه‌داری می‌شوند.

\section{پیاده‌سازی خط لوله‌ی سامانه‌ی واسط}
\label{sec:pipeline-impl}
%----------------------------------------------------------------------

هسته‌ی سامانه یک خط لوله‌ی چهارده‌مرحله‌ای است که هر درخواست \lr{AI} را
از لحظه‌ی ورود تا تحویل پاسخ پردازش می‌کند:

\begin{enumerate}
    \item \textbf{احراز هویت کلید \lr{API}} — تأیید هویت سرویس‌گیرنده با \code{HMAC-SHA256}
    \item \textbf{اعمال لیست سفید \lr{IP}} — بررسی آدرس \lr{IP} در صورت تعریف
    \item \textbf{اعتبارسنجی درخواست} — بررسی فیلدهای اجباری و فرمت
    \item \textbf{محدودیت نرخ} — کنترل تعداد درخواست در دقیقه
    \item \textbf{انتخاب ارائه‌دهنده} — بارگذاری پیکربندی ارائه‌دهنده از \lr{DB}/کش
    \item \textbf{اعتبارسنجی انتخاب ارائه‌دهنده} — تأیید وجود ارائه‌دهنده و مدل در پایگاه داده
    \item \textbf{تخمین مصرف} — برآورد هزینه پیش از ارسال
    \item \textbf{بررسی کیف پول یا اشتراک} — اطمینان از کافی بودن موجودی
    \item \textbf{ارسال درخواست به ارائه‌دهنده} — فراخوانی \lr{HTTP} از طریق آداپتور
    \item \textbf{نرمال‌سازی پاسخ} — تبدیل پاسخ به فرمت یکپارچه
    \item \textbf{اندازه‌گیری مصرف واقعی} — شمارش توکن با توکن‌ساز \lr{BPE}
    \item \textbf{ثبت گزارش} — درج رکورد مصرف در پایگاه داده
    \item \textbf{کسر از کیف پول} — تسویه‌ی هزینه‌ی واقعی
    \item \textbf{ارسال گزارش خارجی} — ارسال به \lr{Loki}/\lr{BetterStack} در صورت پیکربندی
\end{enumerate}

شکل~\ref{fig:chat-completions} نمونه‌ی کامل یک فراخوانی موفق تکمیل گفتگو را
نشان می‌دهد که از تمام مراحل این خط لوله عبور کرده است.

\begin{figure}[H]
    \centering

\section{پیاده‌سازی آداپتورهای ارائه‌دهندگان}
\label{sec:adapters-impl}
%----------------------------------------------------------------------

برای هر ارائه‌دهنده یک آداپتور جداگانه پیاده‌سازی شده که رابط
\code{ProviderAdapterInterface} را پیاده‌سازی می‌کند.

\subsection{\lr{OpenAI}}

آداپتور \lr{OpenAI} به صورت مستقیم با نقاط پایانی استاندارد این شرکت ارتباط
برقرار می‌کند و پشتیبانی کاملی از تمام نقاط پایانی پیاده‌سازی‌شده دارد.

\subsection{\lr{Gemini}}

آداپتور \lr{Gemini} درخواست‌های با فرمت \lr{OpenAI} را به فرمت بومی
\lr{Google Generative AI API} تبدیل کرده و پاسخ را به فرمت استاندارد
نرمال می‌کند.

\subsection{\lr{Groq} و \lr{OpenRouter}}

هر دوی این ارائه‌دهندگان با فرمت \lr{OpenAI} سازگار هستند؛ آداپتور
\lr{OpenAI-Compatible} تنها \lr{Base URL} و کلید \lr{API} آن‌ها را
جایگزین می‌کند.

\section{پیاده‌سازی احراز هویت \lr{OTP}}
\label{sec:otp-impl}
%----------------------------------------------------------------------

جریان احراز هویت سامانه مبتنی بر \term{رمز یک‌بارمصرف}{One-Time Password (OTP)}
است. کاربر با شماره‌ی تلفن یا ایمیل خود درخواست ورود می‌دهد، سامانه یک
رمز شش‌رقمی تولید و ارسال می‌کند، و کاربر پس از دریافت، آن را برای تأیید
هویت ارسال می‌نماید. در صورت تأیید، یک توکن \lr{Sanctum} صادر می‌شود.

شکل~\ref{fig:otp-start} فراخوانی نقطه‌ی پایانی ارسال رمز یک‌بارمصرف را
نشان می‌دهد.

\begin{figure}[H]
    \centering
    \includegraphics[width=0.95\textwidth]{01-start-otp}
    \caption{ارسال درخواست دریافت رمز یک‌بارمصرف}
    \label{fig:otp-start}
\end{figure}

پس از دریافت رمز، کاربر آن را به نقطه‌ی پایانی تأیید ارسال می‌کند و در

\section{پیاده‌سازی گزارش‌دهی و ارسال گزارش خارجی}
\label{sec:reporting-impl}
%----------------------------------------------------------------------

سامانه گزارش‌های تفصیلی مصرف را برای هر کاربر و به تفکیک ارائه‌دهنده
ارائه می‌دهد. این گزارش‌ها شامل تعداد توکن‌های ورودی و خروجی، هزینه،
تأخیر پاسخ و وضعیت هر درخواست هستند. علاوه بر \lr{API} داخلی، سامانه
می‌تواند گزارش‌ها را به سرویس‌های مدیریت گزارش خارجی مانند \lr{Loki}
یا \lr{BetterStack} نیز ارسال کند.

\section{داشبورد \lr{Playground}}
%----------------------------------------------------------------------

داشبورد \lr{Playground} یک رابط کاربری تحت وب است که با \lr{Vue.js}
پیاده‌سازی شده و امکان آزمایش تعاملی تمام نقاط پایانی سامانه را فراهم
می‌کند. کاربر می‌تواند ارائه‌دهنده، مدل و کلید \lr{API} را انتخاب کرده
و درخواست را مستقیماً از مرورگر ارسال نماید.
